DATABASE TABLES – LOANMATE
TABLE – AUTH_USER
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
username STRING (UNIQUE) UNIQUE, NOT NULL
email STRING NOT NULL
password STRING NOT NULL (Hashed – PBKDF2-SHA256)
first_name STRING NOT NULL
last_name STRING NOT NULL
is_staff BOOLEAN DEFAULT FALSE, NOT NULL
is_active BOOLEAN DEFAULT TRUE, NOT NULL
is_superuser BOOLEAN DEFAULT FALSE, NOT NULL
date_joined DATETIME DEFAULT (current timestamp), NOT NULL
last_login DATETIME NULL ALLOWED
TABLE – AUTH_GROUP
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
name STRING (UNIQUE) UNIQUE, NOT NULL (e.g. 'Admins', 'Officers')
TABLE – AUTH_PERMISSION
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
name STRING NOT NULL (Human-readable permission name)
content_type_id INTEGER (FOREIGN KEY) REFERENCES django_content_type(id), NOT NULL
codename STRING NOT NULL (e.g. 'add_user', 'change_user')
TABLE – AUTH_GROUP_PERMISSIONS
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
group_id INTEGER (FOREIGN KEY) REFERENCES auth_group(id), NOT NULL
permission_id INTEGER (FOREIGN KEY) REFERENCES auth_permission(id), NOT NULL
TABLE – AUTH_USER_GROUPS
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
user_id INTEGER (FOREIGN KEY) REFERENCES auth_user(id), NOT NULL
group_id INTEGER (FOREIGN KEY) REFERENCES auth_group(id), NOT NULL
TABLE – AUTH_USER_USER_PERMISSIONS
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
user_id INTEGER (FOREIGN KEY) REFERENCES auth_user(id), NOT NULL
permission_id INTEGER (FOREIGN KEY) REFERENCES auth_permission(id), NOT NULL
TABLE – DJANGO_CONTENT_TYPE
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
app_label STRING NOT NULL (Django app name, e.g. 'loans')
model STRING NOT NULL (Model name, e.g. 'loanapplication')
TABLE – DJANGO_ADMIN_LOG
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
action_time DATETIME NOT NULL (Timestamp of the admin action)
object_id TEXT NULL ALLOWED (PK of the affected object)
object_repr STRING NOT NULL (String representation of object)
action_flag INTEGER NOT NULL (1=Addition, 2=Change, 3=Deletion)
change_message TEXT NOT NULL (JSON description of changes)
content_type_id INTEGER (FOREIGN KEY) REFERENCES django_content_type(id), NULL ALLOWED
user_id INTEGER (FOREIGN KEY) REFERENCES auth_user(id), NOT NULL, ON DELETE CASCADE
TABLE – DJANGO_MIGRATIONS
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
app STRING NOT NULL (App name e.g. 'loans', 'accounts')
name STRING NOT NULL (Migration file name)
applied DATETIME NOT NULL (Timestamp when migration was applied)
TABLE – DJANGO_SESSION
FIELD NAMEDATA TYPEDESCRIPTION
session_key STRING (PRIMARY KEY) PRIMARY KEY, UNIQUE session identifier
session_data TEXT NOT NULL (Encoded session data)
expire_date DATETIME NOT NULL, INDEXED (Session expiry timestamp)
TABLE – ACCOUNTS_USERPROFILE
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
user_id INTEGER (FOREIGN KEY) REFERENCES auth_user(id), UNIQUE, NOT NULL, ON DELETE CASCADE
role STRING DEFAULT 'user', NOT NULL ('user' or 'officer')
phone_number STRING NULL ALLOWED
address TEXT NULL ALLOWED
employment_type STRING NULL ALLOWED
monthly_income INTEGER NULL ALLOWED, POSITIVE
TABLE – LOANS_LOANTYPE
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
name STRING (UNIQUE) UNIQUE, NOT NULL
description TEXT NULL ALLOWED
is_active BOOLEAN DEFAULT TRUE, NOT NULL
TABLE – LOANS_LOANAPPLICATION
FIELD NAMEDATA TYPEDESCRIPTION
id INTEGER (PRIMARY KEY) PRIMARY KEY, AUTO INCREMENT
user_id INTEGER (FOREIGN KEY) REFERENCES auth_user(id), NOT NULL, ON DELETE CASCADE
loan_type_id INTEGER (FOREIGN KEY) REFERENCES loans_loantype(id), NOT NULL, ON DELETE PROTECT
income INTEGER NOT NULL, POSITIVE
employment_type STRING NOT NULL
credit_score INTEGER NOT NULL, POSITIVE
loan_amount INTEGER NOT NULL, POSITIVE
loan_tenure INTEGER NOT NULL, POSITIVE (duration in months)
status STRING DEFAULT 'PENDING', NOT NULL ('PENDING' / 'APPROVED' / 'REJECTED')
remarks TEXT DEFAULT '', NOT NULL
extra_details JSON DEFAULT {}, NULL ALLOWED (ML prediction data)
created_at DATETIME DEFAULT (auto on insert), NOT NULL
reviewed_at DATETIME NULL ALLOWED
reviewed_by_id INTEGER (FOREIGN KEY) REFERENCES auth_user(id), NULL ALLOWED, ON DELETE SET NULL
